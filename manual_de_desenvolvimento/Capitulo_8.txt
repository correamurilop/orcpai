## ğŸ“„ Estrutura da classe `MenuLateral`

```python
import customtkinter as ctk

class MenuLateral(ctk.CTkFrame):
    def __init__(self, master, titulo="Menu", acoes=[]):
        super().__init__(master, width=200, corner_radius=0)
        self.pack_propagate(False)

        self.label = ctk.CTkLabel(self, text=titulo, font=("Arial", 16))
        self.label.pack(pady=10)

        for texto, funcao in acoes:
            botao = ctk.CTkButton(self, text=texto, command=funcao)
            botao.pack(padx=10, pady=5, fill="x")
```
